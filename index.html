Conexión con Google Sheets

Módulos Conectados Exitosamente

El sistema procesa y conecta las tablas SEDES, PROGRAMACIÓN, MANTENIMIENTOS, COMODATO, EQUIPOS, REQUERIMIENTO y DETALLE_REQ para alimentar el Perfil Financiero y Operativo.

Código Recomendado para Google Apps Script (doGet): 
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  function obtenerDatosHoja(nombreHoja) {
    var sheet = ss.getSheetByName(nombreHoja);
    if (!sheet) return [];
    var data = sheet.getDataRange().getValues();
    if (data.length <= 1) return [];
    var headers = data[0];
    var result = [];
    for (var i = 1; i < data.length; i++) {
      var obj = {};
      for (var j = 0; j < headers.length; j++) {
        obj[headers[j]] = data[i][j]; 
      }
      result.push(obj);
    }
    return result;
  }
  var respuesta = {
    programacion: obtenerDatosHoja("PROGRAMACION"),
    mantenimiento: obtenerDatosHoja("MANTENIMIENTO"),
    comodato: obtenerDatosHoja("COMODATO"),
    sedes: obtenerDatosHoja("SEDES"),
    equipos: obtenerDatosHoja("EQUIPOS"), // Habilitado para consultar los costos comodatos unitarios
    requerimiento: obtenerDatosHoja("REQUERIMIENTO"),
    detalle_req: obtenerDatosHoja("DETALLE_REQ")
  };
  return ContentService.createTextOutput(JSON.stringify(respuesta))
    .setMimeType(ContentService.MimeType.JSON);
}
